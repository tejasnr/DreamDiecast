import { v } from "convex/values";

export default defineSchema({

      user: v.string(),

      productId: v.id("products"),

      image: v.string(),
      category: v.string(),
      brand: v.string(),
      scale: v.string(),
      quantity: v.number(),
    })),

    shippingCharges: v.optional(v.number()),

    paymentProofUrl: v.string(),
    paymentMethod: v.string(),

      v.literal("submitted"),
      v.literal("verified"),
      v.literal("rejected")
    ),
    orderStatus: v.union(
      v.literal("pending"),
      v.literal("processing"),
      v.literal("shipped"),
      v.literal("completed"),
      v.literal("cancelled")
    ),
    shippingDetails: v.optional(v.object({
      name: v.string(),
      phone: v.string(),
      address: v.string(),
      city: v.string(),

      pincode: v.string(),
    })),

    .index("by_userId", ["userId"])

  garageItems: defineTable({
    userId: v.id("users"),

    name: v.string(),
    price: v.number(),
    originalPrice: v.optional(v.number()),

    category: v.string(),
    brand: v.string(),
    scale: v.string(),
    purchasedAt: v.number(),

      v.literal("arrived")
    ),
  })
    .index("by_userId", ["userId"])
    .index("by_userId_status", ["userId", "status"]),

    userId: v.id("users"),
    productId: v.id("products"),
    name: v.string(),
    price: v.number(),
    image: v.string(),
    category: v.string(),

    scale: v.string(),

    key: v.string(),
    heroBackground: v.optional(v.string()),
    vaultImage: v.optional(v.string()),
    footerBackground: v.optional(v.string()),
    categoryJdm: v.optional(v.string()),

    categoryHypercars: v.optional(v.string()),

  assets: defineTable({
    name: v.string(),
    url: v.string(),
    storageId: v.optional(v.id("_storage")),
    type: v.union(v.literal("image"), v.literal("video")),
  }),

### 2.2 Auth Flow

1. User clicks "Sign In" -> redirect to WorkOS AuthKit
2. WorkOS handles Google OAuth (and optionally email/password)
3. Callback at `/api/auth/callback` receives auth code
4. Exchange code for WorkOS session + user profile
5. Look up or create user in Convex `users` table
6. Store session token in HTTP-only cookie
7. `AuthContext` reads session and provides user to app

## 9. Edge Cases & Error Handling

- **Convex offline/errors**: Convex auto-retries queries. Show toast on mutation failures.
- **WorkOS callback failure**: Redirect to homepage with error toast if OAuth fails.
- **PostHog ad-blockers**: PostHog gracefully degrades â€” no user-facing errors. Optionally proxy through `/api/posthog/` to bypass blockers.
- **Migration data integrity**: Validate all Firestore document IDs are mapped correctly. Product references in orders/garage must point to valid Convex IDs post-migration.
- **Image URL migration**: Firebase Storage URLs will break after Firebase teardown. Must re-upload all images to Convex storage and update URLs in all documents.
- **Admin role sync**: Ensure the three admin emails are seeded in Convex `users` table with `role: "admin"` during migration.
- **WhatsApp/Instagram links**: Both open in new tab. If WhatsApp deep link fails on desktop, WhatsApp Web handles it automatically.
